# Pathfinder Neo4j schema — reference for agents and Cypher generation
# Neo4j 5.x, labels and keys below are authoritative.

## Nodes
# (:Entity {id: string, name: string, type: string, influence: float, sentiment: float})
# (:System {id: string, name: string})
# Use Entity for Person, Role, Agency, Gatekeeper; type property disambiguates.

## Relationships
# (:Entity)-[:REPORTS_TO]->(:Entity)
# (:Entity)-[:INFLUENCES {weight: float}]->(:Entity)
# (:Entity)-[:BLOCKS {reason: string}]->(:Entity)
# (:Entity)-[:USES {telemetry_score: float}]->(:System)
# (:Entity)-[:LINKS_TO]->(:Entity)   # from Obsidian [[wiki]] when type unknown

## Constraints (run once)
CREATE CONSTRAINT entity_id IF NOT EXISTS FOR (e:Entity) REQUIRE e.id IS UNIQUE;
CREATE CONSTRAINT system_id IF NOT EXISTS FOR (s:System) REQUIRE s.id IS UNIQUE;
